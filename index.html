import httpx import asyncio from typing import Dict, Any from dataclasses import dataclass @dataclass class MtgCard: name: str mana_cost: str type_line: str oracle_text: str class ScryfallClient: def __init__(self): self.base_url = "https://api.scryfall.com" self.client = httpx.AsyncClient() async def get_sol_ring(self) -> MtgCard: """Fetch Sol Ring card data from Scryfall API""" url = f"{self.base_url}/cards/named?exact=sol+ring" response = await self.client.get(url) response.raise_for_status() data = response.json() return MtgCard( name=data["name"], mana_cost=data.get("mana_cost", ""), type_line=data["type_line"], oracle_text=data.get("oracle_text", "") ) async def close(self): await self.client.aclose() # Usage example async def main(): client = ScryfallClient() sol_ring = await client.get_sol_ring() print(f"{sol_ring.name} - {sol_ring.type_line}") await client.close() asyncio.run(main())
